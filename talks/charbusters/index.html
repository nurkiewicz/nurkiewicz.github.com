int graphemes(String str) {
   BreakIterator graphemeCounter = BreakIterator
           .getCharacterInstance(Locale.US);
   graphemeCounter.setText(str);
   int graphemeCount = 0;

       graphemeCount++;
   }
   return graphemeCount;
}
